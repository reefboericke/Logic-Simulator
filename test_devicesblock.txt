begin devices:
	AND G1: inputs = 2;
	NAND G2: inputs = 2;
	# implement a NOT gate using a two input NAND #
	NAND notS: inputs = 2;
	OR G3: inputs = 2;
	SWITCH A: initial = 0;
	SWITCH B: initial = 0;
	SWITCH S: initial = 0;
    DTYPE D;
	DTYPE D1;
end devices;

begin connections:
	A -> G1.I1;
	S -> notS.I1;
	S -> notS.I2;
	notS -> G1.I2;
	S -> G2.SET;
	B -> D.I1;
	G1 -> G3.I1;
    D.QBAR -> G3.I2;
end connections;

begin monitors:
	G1; 
	G3; 
	G3;
end monitors;
